Extension TypeScript Code (src/extension.ts)

import * as vscode from 'vscode';
import axios from 'axios';

let baseUrl = 'http://127.0.0.1:8155';
const HF_TOKEN = 'hf_YOUR_TOKEN_HERE';

export function activate(context: vscode.ExtensionContext) {
  const config = vscode.workspace.getConfiguration('zombiecoder');
  baseUrl = config.get('serverUrl', 'http://127.0.0.1:8155');
  
  // Set HuggingFace token on activation
  axios.post(`${baseUrl}/auth/hf_token`, {token: HF_TOKEN}).catch(() => {});

  // Load Model Command
  context.subscriptions.push(
    vscode.commands.registerCommand('zombiecoder.loadModel', async () => {
      const models = await getInstalledModels();
      const selected = await vscode.window.showQuickPick(models);
      if (selected) {
        await loadModel(selected);
      }
    })
  );

  // Generate Text Command
  context.subscriptions.push(
    vscode.commands.registerCommand('zombiecoder.generateText', async () => {
      const editor = vscode.window.activeTextEditor;
      if (!editor) return;
      
      const selection = editor.selection;
      const prompt = editor.document.getText(selection) || 
                     await vscode.window.showInputBox({ prompt: 'Enter prompt' });
      
      if (prompt) {
        const model = config.get('defaultModel', 'tinyllama-gguf');
        const result = await generate(model, prompt);
        editor.edit(editBuilder => {
          editBuilder.replace(selection, result);
        });
      }
    })
  );

  // Unload Model Command
  context.subscriptions.push(
    vscode.commands.registerCommand('zombiecoder.unloadModel', async () => {
      const runningModels = await getRunningModels();
      const selected = await vscode.window.showQuickPick(runningModels);
      if (selected) {
        await unloadModel(selected);
      }
    })
  );
}

async function getInstalledModels(): Promise {
  const res = await axios.get(`${baseUrl}/models/installed`);
  return res.data.models.map((m: any) => m.name);
}

async function getRunningModels(): Promise {
  const res = await axios.get(`${baseUrl}/runtime/status`);
  return res.data.models.map((m: any) => m.model);
}

async function loadModel(model: string) {
  await vscode.window.withProgress({
    location: vscode.ProgressLocation.Notification,
    title: `Loading ${model}...`,
    cancellable: false
  }, async () => {
    await axios.post(`${baseUrl}/runtime/load/${model}?threads=4`);
    vscode.window.showInformationMessage(`‚úÖ ${model} loaded`);
  });
}

async function generate(model: string, prompt: string): Promise {
  const res = await axios.post(`${baseUrl}/api/generate`, {
    model,
    prompt,
    stream: false
  });
  return res.data.response || res.data.content || JSON.stringify(res.data);
}

async function unloadModel(model: string) {
  await axios.post(`${baseUrl}/runtime/unload/${model}`);
  vscode.window.showInformationMessage(`‚èπÔ∏è ${model} unloaded`);
}

export function deactivate() {}
